claim(host=nil, pid=nil)

Returns the next Job to be processed, if there is one available. Nil will be returned if a Job already is being processed (processing is strictly sequential and pipelined) or no Job is available.

[Source]

     # File lib/job_queue.rb, line 222
222:   def claim(host=nil, pid=nil)
223:     self.db.transaction do
224:       record_worker_in_touch(host, pid) if host && pid
225:       job = nil
226:       if db[:processing_ordered]
227:         # If there is an ordered job being processed, choose an unordered job if there is one
228:         waiting_unordered = db[:waiting_unordered]
229:         return nil if waiting_unordered.length == 0
230:         job = db[waiting_unordered.first]
231:         db[:waiting_unordered] = waiting_unordered[1..-1]
232:         db[:processing_unordered] << job.uuid
233:       elsif ((waiting_ordered = db[:waiting_ordered]) != [])
234:         # If no ordered job is being processed, but there is an ordered job waiting, choose it
235:         job = db[waiting_ordered.first]
236:         db[:waiting_ordered] = waiting_ordered[1..-1]
237:         db[:processing_ordered] = job.uuid
238:       elsif ((waiting_unordered = db[:waiting_unordered]) != [])
239:         # If no ordered jobs were waiting, choose an unordened one, if available
240:         job = db[waiting_unordered.first]
241:         db[:waiting_unordered] = waiting_unordered[1..-1]
242:         db[:processing_unordered] << job.uuid
243:       end
244:       # Have we selected a job? If so, adorn it
245:       return nil unless job
246:       job.begun_at = Time.now.utc
247:       db[job.uuid] = job
248:       job
